These are test cases for 1-my_list.py
-------------------------------------------

case 0: importing
>>> MyList = __import__('1-my_list').MyList
>>>
-------------------------------------------

case 1: common case

>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]
>>>

-------------------------------------------

case 2: For un defined object

>>> MyList = __import__('1-my_list').MyList
>>> my_list2.print_sorted()

  File "<stdin>", line 1, in <module>
NameError: name 'my_list2' is not defined
>>>

-------------------------------------------

case 3: If a list is passed

>>> a = [2, 10, 8]
>>> my_list = MyList(a)
>>> my_list.print_sorted()
[2, 8, 10]
>>>

-------------------------------------------

case 4: If an empty list is given

>>> a = []
>>> my_list = MyList(a)
>>> my_list.print_sorted()
[]
>>>

-------------------------------------------

case 5: If an unique number is used

>>> my_list = MyList()
>>> my_list.append(3)
>>> my_list.print_sorted()
[3]
>>> print(my_list)
[3]
>>>

-------------------------------------------

case 6: If same numbers are used

>>> my_list2 = MyList()
>>> my_list2.append(8)
>>> my_list2.append(8)
>>> my_list2.append(8)
>>> my_list2.append(8)
>>> my_list2.print_sorted()
[8, 8, 8, 8]
>>>

-------------------------------------------

case 7: If parent object
>>> my_list = [3, 4, 5]

  File "<stdin>", line 1, in <module>
AttributeError: 'list' object has no attribute 'print_sorted'
>>>

-------------------------------------------

case 8: If negative numbers are used

>>> MyList = __import__('1-my_list').MyList
>>> my_list = MyList()

>>> my_list.append(3)
>>> my_list.append(-11)
